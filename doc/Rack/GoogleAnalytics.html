def initialize(app, options = nil)
  raise ArgumentError, "Tracker must be set!" unless options[:tracker] and !options[:tracker].empty?
  options       = DEFAULT_OPTIONS.merge(options)
  @app          = app
  @tracker      = options[:tracker]
  @environments = options[:environments]
end

def current_environment
  KNOWN_ENVIRONMENTAL_VARIABLES.push('check_the_rails_constant').each do |env|
    return ENV[env] if ENV.has_key?(env)
    if env == 'check_the_rails_constant'
      return false unless defined?(Rails)
      return Rails.env
    end
   end
   return false # oopse :(  
end
